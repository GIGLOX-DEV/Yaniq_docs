// Get logger instance
private final CommonLogger logger = CommonLogger.getLogger();

// Basic logging with placeholders
logger.info("User {} authenticated successfully", userId);

logger.security("LOGIN_SUCCESS", correlationId, "User: {}", maskedEmail);

Quick Start Guide

Installation

XML 
<dependency>
    <groupId>com.yaniq</groupId>
    <artifactId>common-logging</artifactId>
    <version>1.0.0</version>
</dependency>

Basic Usage

Java 
import com.yaniq.common_logging.CommonLogger;

@Service
public class MyService {
    private final CommonLogger logger = CommonLogger.getLogger();

    public void processUser(String userId) {
        logger.info("Processing user: {}", userId);

        try {
            // Business logic here
            logger.info("User {} processed successfully", userId);
        } catch (Exception e) {
            logger.error("Failed to process user: {}", userId, e);
        }
    }
}

Configuration

Environment Configuration

Set the LOG_MODE environment variable to choose your logging strategy:

Bash 
# Console output (default)
export LOG_MODE=CONSOLE

# ELK Stack compatible
export LOG_MODE=ELK

# File-based logging
export LOG_MODE=FILE

# Structured JSON
export LOG_MODE=STRUCTURED

Usage Guide

Logger Factory Methods

CommonLogger.getLogger()

Get the default logger instance

CommonLogger logger = CommonLogger.getLogger();

CommonLogger.getLogger(Class<?> clazz)

Get a class-specific logger instance

CommonLogger logger = CommonLogger.getLogger(MyService.class);

Basic Logging Methods

Java 
// Different log levels
logger.trace("Detailed execution trace");
logger.debug("Debug information for: {}", userId);
logger.info("User {} logged in successfully", userId);
logger.warn("Rate limit approaching for user: {}", userId);
logger.error("Database connection failed", exception);
logger.fatal("Critical system failure occurred");

Specialized Logging

Security Event Logging

Track security-related events with correlation IDs

logger.security("LOGIN_ATTEMPT", correlationId, "IP: {} | UserAgent: {}", ip, userAgent);
logger.security("UNAUTHORIZED_ACCESS", correlationId, "User: {} | Resource: {}", userId, resource);

Performance Monitoring

Track operation performance and timing

logger.performance("database_query", duration, "Table: {} | Records: {}", table, count);
logger.performance("api_call", responseTime, "Endpoint: {} | Status: {}", endpoint, status);

Audit Logging

Create audit trails for compliance

logger.audit("USER_CREATE", adminId, "Created user: {} | Role: {}", newUserId, role);
logger.audit("DATA_DELETE", userId, "Deleted records: {} | Reason: {}", count, reason);

Configuration

Environment Variables

Variable Default Description
LOG_MODE CONSOLE Logging strategy: CONSOLE, ELK, FILE, STRUCTURED
service.name yaniq-service Service name for ELK logs
yaniq.log.dir logs Log directory for file strategy
yaniq.log.debug false Enable debug level logging

Docker Configuration

YAML 
version: '3.8'
services:
  auth-service:
    image: yaniq/auth-service:1.0.0
    environment:
      - LOG_MODE=ELK
      - service.name=auth-service
      - yaniq.log.debug=false
    volumes:
      - ./logs:/var/logs/yaniq

Kubernetes Configuration

YAML 
apiVersion: v1
kind: ConfigMap
metadata:
  name: logging-config
data:
  LOG_MODE: "ELK"
  service.name: "auth-service"
  yaniq.log.debug: "false"

API Reference

Core Methods

info(String format, Object... args)

Log an info level message with placeholder support

logger.info("User {} authenticated in {}ms", userId, duration);

error(String format, Throwable throwable, Object... args)

Log an error with exception and structured data

logger.error("Database operation failed for user: {}", userId, exception);

Specialized Methods

security(String eventType, String correlationId, String format, Object... args)

Log security events with correlation tracking

logger.security("LOGIN_FAILURE", correlationId, "User: {} | Reason: {}", email, reason);

performance(String operation, long duration, String format, Object... args)

Log performance metrics for monitoring

logger.performance("user_lookup", 150, "Database: {} | Cache: {}", dbName, cacheHit);

metric(String metricName, Object value, String... tags)

Log application metrics with tags

logger.metric("active_users", userCount, "service:auth", "region:us-east");

Auth Service Integration

Authentication Service Logging

Best practices for logging in authentication workflows

Java 
@Service
public class AuthenticationService {
    private final CommonLogger logger = CommonLogger.getLogger(AuthenticationService.class);

    public LoginResponse authenticate(LoginRequest request) {
        String correlationId = UUID.randomUUID().toString().substring(0, 8);
        String maskedEmail = maskEmail(request.getEmail());

        logger.info("AUTH_START | CorrelationId: {} | Email: {}", correlationId, maskedEmail);

        try {
            // Authentication logic
            logger.security("LOGIN_SUCCESS", correlationId, "User: {}", maskedEmail);
            return response;
        } catch (Exception e) {
            logger.security("LOGIN_FAILURE", correlationId, "User: {} | Error: {}",
                maskedEmail, e.getMessage());
            throw e;
        }
    }
}

Validation Service Integration

Java 
@Service
public class ValidationService {
    private final CommonLogger logger = CommonLogger.getLogger(ValidationService.class);

    public void validateEmail(String email) {
        logger.debug("Validating email format");

        if (email == null || email.trim().isEmpty()) {
            logger.warn("VALIDATION_FAILED | Field: email | Reason: null or empty");
            throw new ValidationException("Email is required");
        }

        logger.debug("Email validation passed for: {}", maskEmail(email));
    }
}
